## Install

```bash
npm install --save react-date-input
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-date-input'
import 'react-date-input/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
